11  package com.gvmax.server.handlers;
12  
13  import java.io.IOException;
14  
15  import org.apache.log4j.Logger;
16  
17  import com.gvmax.common.model.Email;
18  import com.gvmax.common.model.User;
19  import com.gvmax.common.util.EmailUtils;
20  import com.gvmax.data.user.UserDAO;
21  import com.gvmax.server.util.Notifier;
22  
23  /**
24   * This class handles sms send emails. Both {pin}-{destination}@xxxx and
25   * {pin}-{destination}-{reply}@xxxxx
26   */
27  public class EmailInHandler {
28      /** Logger */
29      private static final Logger logger = Logger.getLogger(EmailInHandler.class);
30      /** User DAO */
31      private UserDAO userDAO;
32  
33      // Extracted info
34      private User user;
35      private String destination;
36      private String text;
37  
38      public EmailInHandler(UserDAO userDAO) {
39          this.userDAO = userDAO;
40      }
41  
42      public void handle(Email email) {
43          extractInfo(email);
44          if (user == null) {
45              logger.debug("Unable to retrieve user from email: " + email.getTo());
46              return;
47          }
48          userDAO.incrementEmailInCount(user.getEmail());
49          if (!user.isGvPassword()) {
50              logger.debug("Non GV user attempted to send SMS via email");
51              userDAO.incrementInvalidEmailCount(user.getEmail());
52              return;
53          }
54  
55          Notifier notifier = new Notifier(user, userDAO, null);
56          try {
57              notifier.sendSMS(destination, text);
58          } catch (IOException e) {
59              userDAO.incrementErrorCount(user.getEmail());
60              logger.warn("Unable to send sms via email : " + e.getMessage());
61          }
62  
63      }
64  
65      private void extractInfo(Email email) {
66          String recipient = email.getTo().substring(0, email.getTo().indexOf('@'));
67  
68          // Get user
69          if (recipient.indexOf('-') == -1) {
70              return;
71          }
72  
73          String pin = recipient.substring(0, recipient.indexOf('-'));
74          user = userDAO.retrieveByPin(pin);
75          if (user == null) {
76              return;
77          }
78  
79          // Get destination
80          destination = recipient.substring(pin.length() + 1);
81          if (destination.endsWith("-reply")) {
82              destination = destination.substring(0, destination.lastIndexOf('-'));
83          }
84  
85          text = EmailUtils.strip(email.getText());
86  
87      }
88  
89  }
