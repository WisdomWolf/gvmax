30  public class SMTPServerImpl {

32      private SMTPServer server;
33      private QueueDAO<Email> smtpQueue;

35      public SMTPServerImpl(int port, QueueDAO<Email> smtpQueue) {
36          this.smtpQueue = smtpQueue;
37          server = new SMTPServer(new SimpleMessageListenerAdapter(new SMTPListener()));
38          server.setPort(port);
39          server.setHideTLS(true);
40      }

42      public void start() {
43          server.start();
44          logger.info("SMTPServerImpl started on port: " + server.getPort());
45      }
46  
47      public void stop() {
48          server.stop();
49          logger.info("SMTPServerImpl at port: " + server.getPort() + " stopped.");
50      }

52      class SMTPListener implements SimpleMessageListener {
53  
54          @Override
55          public void deliver(String from, String recipient, InputStream data) throws IOException {
56              logger.debug("email received from:" + from + " to:" + recipient);
57              MetricsUtil.getCounter(SMTPListener.class,"received").inc();
58              // Extract Email
59              Email email = null;
60              try {
61                  email = extractEmail(from, recipient, data);
62              } catch (Exception e) {
63                  logger.error("error processing email from: " + from + " error: " + e.getMessage(), e);
64                  MetricsUtil.getCounter(SMTPListener.class,"error").inc();
65                  return;
66              }
67              if (from.contains("thehealingmind")) {
68                  logger.warn("rejected: " + from + " TEXT =\n" + email.getText());
69                  MetricsUtil.getCounter(SMTPListener.class,"rejected").inc();
70                  return;
71              }
72              if (from.endsWith("amazonaws.com")) {
73                  logger.warn("rejected: " + from + " TEXT =\n" + email.getText());
74                  MetricsUtil.getCounter(SMTPListener.class,"rejected").inc();
75                  throw new TooMuchDataException("rejected");
76              }
77              // Queue email
78              MetricsUtil.getCounter(SMTPListener.class,"delivered").inc();
79              smtpQueue.enqueue(email);
80          }
81  
82          private Email extractEmail(String from, String recipient, InputStream data) throws IOException, MessagingException {
83              Email email = new Email();
84              email.setTimestamp(System.currentTimeMillis());
85              email.setFrom(from);
86              email.setTo(recipient);
87              // Create message
88              MimeMessage msg = new MimeMessage(null, data);
89              // Extract message text
90              if (msg.getContent() instanceof MimeMultipart) {
91                  MimeMultipart t = (MimeMultipart) msg.getContent();
92                  email.setText(t.getBodyPart(0).getContent().toString());
93              } else {
94                  email.setText(msg.getContent().toString());
95              }
96  
97              String[] froms = msg.getHeader("From");
98              if (froms != null && froms.length > 0) {
99                  email.setOriginalFrom(froms[0]);
100             }
101 
102             email.setSubject(msg.getSubject());
103 
104             return email;
105         }

107         @Override
108         public boolean accept(String from, String recipient) {
109             boolean ok = recipient.toLowerCase().endsWith("my.gvmax.com");
110             if (!ok) {
111                 logger.warn("rejected, from: " + from + " to: " + recipient);
112             }
113             return ok;
114         }
115     }
