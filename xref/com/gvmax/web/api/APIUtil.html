20  import com.gvmax.common.model.User;

24  public final class APIUtil {
25      private static final Logger logger = Logger.getLogger(APIUtil.class);
26  
27      private APIUtil() {}
28  
29      public static User getUser(GVMaxServiceImpl service, HttpSession session, String pin) {
30          return service.getUser((String) session.getAttribute("email"), pin);
31      }
32  
33      public static void invalidCredentials(HttpServletResponse resp) {
34          logger.warn("Invalid credentials");
35          sendError(403, "Invalid credentials", resp);
36      }
37  
38      public static void internalError(Exception e, HttpServletResponse resp) {
39          logger.warn("Internal Error: " + e.getMessage());
40          sendError(500, "Internal Error : " + e.getMessage(), resp);
41      }
42  
43      public static void sendError(int code, String error, HttpServletResponse resp) {
44          try {
45              resp.getWriter().println(error);
46              resp.setStatus(code);
47          } catch (IOException e) {
48              MiscUtils.emptyBlock();
49          }
50      }
