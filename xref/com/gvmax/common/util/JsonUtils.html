11  package com.gvmax.common.util;

13  import java.io.IOException;
14  import java.text.SimpleDateFormat;
15  import java.util.List;
16  
17  import com.fasterxml.jackson.annotation.JsonInclude.Include;
18  import com.fasterxml.jackson.databind.DeserializationFeature;
19  import com.fasterxml.jackson.databind.JsonNode;
20  import com.fasterxml.jackson.databind.ObjectMapper;
21  import com.fasterxml.jackson.databind.SerializationFeature;
22  import com.fasterxml.jackson.databind.type.CollectionType;
23  
24  /**
25   * Utility class used to convert objects to / from Json.
26   */
27  public final class JsonUtils {
28      /** Date formater used to format dates */
29      public static final String DATE_FORMAT = "MMM dd, yyyy hh:mm:ss a";
30      /** Jackson mapper used under covers */
31      private static ObjectMapper mapper = new ObjectMapper();
32      /** Jackson mapper used under covers (pretty print)*/
33      private static ObjectMapper prettyMapper = new ObjectMapper();
34  
35      /**
36       * Utility hidding constructor.
37       */
38      private JsonUtils() {}
39  
40      static {
41          // Init mappers
42          mapper.setSerializationInclusion(Include.NON_NULL);
43          mapper.setDateFormat(new SimpleDateFormat(DATE_FORMAT));
44          mapper.configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);
45  
46          prettyMapper.setSerializationInclusion(Include.NON_NULL);
47          prettyMapper.setDateFormat(new SimpleDateFormat(DATE_FORMAT));
48          prettyMapper.configure(SerializationFeature.INDENT_OUTPUT, true);
49          prettyMapper.configure(DeserializationFeature.FAIL_ON_UNKNOWN_PROPERTIES, false);
50      }
51  
52      /**
53       * Converts an object into a json string.
54       *
55       * @param obj
56       *            The object to convert to json. If null this method returns
57       *            null.
58       * @return Json string representation of object
59       * @throws IOException
60       *             Can throw IOExceptions
61       */
62      public static String toJson(Object obj) throws IOException {
63          return toJson(obj, false);
64      }
65  
66      public static String toJson(Object obj, boolean prettyPrint) throws IOException {
67          if (obj == null) {
68              return null;
69          }
70          if (prettyPrint) {
71              return prettyMapper.writeValueAsString(obj);
72          }
73          return mapper.writeValueAsString(obj);
74      }
75  
76      public static String format(String json, boolean pretty) {
77          try {
78              JsonNode node = fromJson(json, JsonNode.class);
79              return toJson(node, pretty);
80          } catch (Exception e) {
81              return json;
82          }
83      }
84  
85      /**
86       * Converts a json string into an object. If any of the parameters are null
87       * this method returns null.
88       *
89       * @param json
90       *            The json string.
91       * @param valueType
92       *            The required object type.
93       * @return The object or null
94       * @throws IOException
95       *             Can throw IOExceptions
96       */
97      public static <T> T fromJson(String json, Class<T> valueType) throws IOException {
98          if (json == null || valueType == null) {
99              return null;
100         }
101         try {
102             return mapper.readValue(json, valueType);
103         } catch (Exception e) {
104             throw new IOException("error converting json to '" + valueType.getCanonicalName() + "', json=" + json + "'", e);
105         }
106     }
107 
108     // -----------------
109     // COLLECTIONS
110     // -----------------
111 
112     public static <T> List<T> fromJsonList(String json, Class<T> elementClass) throws IOException {
113         if (json == null || elementClass == null) {
114             return null;
115         }
116         CollectionType type = mapper.getTypeFactory().constructCollectionType(List.class, elementClass);
117         return mapper.readValue(json, type);
118     }
