24   * IO related utilities.
25   * Based on commons-io but stripped down to what I need.

27  public final class IOUtil {

32      private IOUtil() {}

34      public static void close(Closeable stream) {
35          if (stream == null) {
36              return;
37          }
38          try {
39              stream.close();
40          } catch (IOException e) {
41              MiscUtils.emptyBlock();
42          }
43  
44      }

46      public static byte[] toByteArray(InputStream in) throws IOException {
47          ByteArrayOutputStream out = new ByteArrayOutputStream();
48          int read = 0;
49          byte[] buffer = new byte[1024];
50          while (read != -1) {
51              read = in.read(buffer);
52              if (read != -1) {
53                  out.write(buffer, 0, read);
54              }
55          }
56          out.close();
57          return out.toByteArray();
58      }
59  
60      public static String toString(InputStream in) throws IOException {
61          return toString(in, Constants.UTF8);
62      }
63  
64      public static String toString(InputStream in, String charset) throws IOException {
65          StringWriter out = new StringWriter();
66          copy(new InputStreamReader(in, charset), out);
67          out.close();
68          in.close();
69          return out.toString();
70      }
71  
72      public static int copy(InputStream in, OutputStream out) throws IOException {
73          int count = 0;
74          int read = 0;
75          byte[] buffer = new byte[1024];
76          while (read != -1) {
77              read = in.read(buffer);
78              if (read != -1) {
79                  count += read;
80                  out.write(buffer, 0, read);
81              }
82          }
83          return count;
84      }
85  
86      public static int copy(Reader input, Writer output) throws IOException {
87          char[] buffer = new char[1024];
88          int count = 0;
89          int n = 0;
90          while (-1 != (n = input.read(buffer))) {
91              output.write(buffer, 0, n);
92              count += n;
93          }
94          return count;
95      }
