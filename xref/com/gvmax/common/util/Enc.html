13  import java.security.Key;
14  import java.security.spec.KeySpec;

16  import javax.crypto.Cipher;
17  import javax.crypto.SecretKeyFactory;
18  import javax.crypto.spec.IvParameterSpec;
19  import javax.crypto.spec.PBEKeySpec;
20  import javax.crypto.spec.SecretKeySpec;

22  import org.apache.commons.codec.binary.Base64;
23  import org.apache.log4j.Logger;

25  /**
26   * Utility used to encrypt and decrypt values.
27   * If the provided password isBlank than encryption is disabled.
28   */
29  public class Enc {
30      private static final Logger logger = Logger.getLogger(Enc.class);
31      /** Is encruption enabled */
32      private boolean enabled;
33      /** Key to use */
34      private Key key;
35      /** Cipher to use */
36      private Cipher c;
37  
38      private static byte[] salt = { (byte) 0x19, (byte) 0x73, (byte) 0x41, (byte) 0x8c, (byte) 0x7e, (byte) 0xd8, (byte) 0xee, (byte) 0x89 };
39      private static byte[] iv = { (byte) 0x19, (byte) 0x73, (byte) 0x41, (byte) 0x8c, (byte) 0x7e, (byte) 0xd8, (byte) 0xee, (byte) 0x89, (byte) 0x19, (byte) 0x73, (byte) 0x41, (byte) 0x8c, (byte) 0x7e, (byte) 0xd8, (byte) 0xee, (byte) 0x89 };
40  
41      // --------------
42      // CONSTRUCTORS
43      // --------------
44  
45      public Enc(String password) {
46          this(password, 256);
47      }
48  
49      public Enc(String password, int keyLength) {
50          if (password == null || password.trim().equals("")) {
51              enabled = false;
52          } else {
53              enabled = true;
54              try {
55                  SecretKeyFactory factory = SecretKeyFactory.getInstance("PBKDF2WithHmacSHA1");
56                  KeySpec spec = new PBEKeySpec(password.toCharArray(), salt, 1024, keyLength);
57                  key = new SecretKeySpec(factory.generateSecret(spec).getEncoded(), "AES");
58                  c = Cipher.getInstance("AES/CBC/PKCS5Padding");
59              } catch (Exception e) {
60                  logger.error(e);
61              }
62          }
63      }
64  
65      // ------------
66      // ENC/DEC
67      // ------------
68  
69      public String encrypt(String valueToEnc) {
70          if (!enabled) {
71              return valueToEnc;
72          }
73          if (valueToEnc == null) {
74              return null;
75          }
76          try {
77              c.init(Cipher.ENCRYPT_MODE, key, new IvParameterSpec(iv));
78              byte[] encValue = c.doFinal(valueToEnc.getBytes("UTF-8"));
79              return Base64.encodeBase64String(encValue);
80          } catch (Exception e) {
81              logger.warn("Unable to encrypt: "+valueToEnc,e);
82              return null;
83          }
84      }
85  
86      public String decrypt(String encryptedValue) {
87          if (!enabled) {
88              return encryptedValue;
89          }
90          byte[] val = decryptByte(encryptedValue);
91          if (val == null) {
92              return null;
93          }
94          return StringUtil.toString(val);
95      }
96  
97      public byte[] decryptByte(String encryptedValue) {
98          if (encryptedValue == null) {
99              return null;
100         }
101         if (!enabled) {
102             return StringUtil.getBytes(encryptedValue);
103         }
104         try {
105             c.init(Cipher.DECRYPT_MODE, key, new IvParameterSpec(iv));
106             byte[] decordedValue = Base64.decodeBase64(encryptedValue);
107             return c.doFinal(decordedValue);
108         } catch (Exception e) {
109             logger.warn("Unable to decrypt: "+encryptedValue,e);
110             return null;
111         }
112     }
