22  /**
23   * Utility used to send howl notifications.
24   */
25  public final class Howl {
26  
27      private Howl() {}
28  
29      public static void notify(String username, String password, String application, String title, String description, String hostname, String iconName) throws IOException {
30          Map<String, String> params = new HashMap<String, String>();
31          params.put("name", "GVMax");
32          params.put("application", application);
33          params.put("title", title);
34          params.put("description", description);
35          params.put("icon-name", iconName);
36          params.put("hostname", "www.gvmax.com");
37  
38          String userPassword = username + ":" + password;
39          String encoded = StringUtil.toString(Base64.encodeBase64(StringUtil.getBytes(userPassword)));
40          Map<String, String> properties = new HashMap<String, String>();
41          properties.put("Authorization", "Application " + encoded);
42  
43          NetUtil.doPost("https://howlapp.com/public/api/notification", params, properties);
44      }
45  
46  }
