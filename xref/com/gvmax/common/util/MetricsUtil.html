33  /**
34   * Metrics related utilities.
35   */
36  public final class MetricsUtil {
37      private static MetricRegistry registry = new MetricRegistry();
38      /** Created timers */
39      private static Map<String, Timer> timers = new HashMap<String, Timer>();
40      /** Created counters */
41      private static Map<String, Counter> counters = new HashMap<String, Counter>();
42      /** Created meters */
43      private static Map<String, Meter> meters = new HashMap<String, Meter>();
44  
45      static {
46          registry.register(MetricRegistry.name("jvm", "gc"), new GarbageCollectorMetricSet());
47          registry.register(MetricRegistry.name("jvm", "memory"), new MemoryUsageGaugeSet());
48          registry.register(MetricRegistry.name("jvm", "thread-states"), new ThreadStatesGaugeSet());
49          registry.register(MetricRegistry.name("jvm", "fd", "usage"), new FileDescriptorRatioGauge());
50  
51          InstrumentedAppender appender = new InstrumentedAppender(registry);
52          appender.activateOptions();
53          LogManager.getRootLogger().addAppender(appender);
54  
55          JmxReporter.forRegistry(registry).inDomain("metrics.gvmax").filter(new MetricFilter() {
56              @Override
57              public boolean matches(String name, Metric metric) {
58                  return name.contains("gvmax");
59              }
60          }).build().start();
61          JmxReporter.forRegistry(registry).inDomain("metrics.jvm").filter(new MetricFilter() {
62              @Override
63              public boolean matches(String name, Metric metric) {
64                  return name.startsWith("jvm.");
65              }
66          }).build().start();
67          JmxReporter.forRegistry(registry).inDomain("metrics.jetty").filter(new MetricFilter() {
68              @Override
69              public boolean matches(String name, Metric metric) {
70                  return name.contains("jetty");
71              }
72          }).build().start();
73          JmxReporter.forRegistry(registry).inDomain("metrics.log4j").filter(new MetricFilter() {
74              @Override
75              public boolean matches(String name, Metric metric) {
76                  return name.contains("log4j");
77              }
78          }).build().start();
79          JmxReporter.forRegistry(registry).inDomain("metrics.all").build().start();
80      }
81  
82      private MetricsUtil() {}

90      public static Timer getTimer(Class<?> clazz,String name) {
91          return getTimer(clazz.getCanonicalName(), name);
92      }
93  
94      public static Timer getTimer(String className, String name) {
95          synchronized (timers) {
96              String key = name(className,name);
97              Timer timer = timers.get(key);
98              if (timer == null) {
99                  timer = registry.timer(key);
100                 timers.put(key, timer);
101             }
102             return timer;
103         }
104     }
