24  /**
25   * Client library used to talk to GVMax Relay.

27  public class GVMaxRelay {
28      private static final Logger logger = Logger.getLogger(GVMaxRelay.class);
29      private String appName;
30      private String host;
31      private String encKey;
32  
33      private String emailHost;
34      private String emailSender;

36      public GVMaxRelay() {}

38      public GVMaxRelay(String host, String encKey, String emailHost, String emailSender) {
39          this.appName = host.split("//", 2)[1].split("\\.", 2)[0];
40          this.host = host;
41          this.encKey = encKey;
42          this.emailHost = emailHost;
43          this.emailSender = emailSender;
44      }
45  
46      public String getAppName() {
47          return appName;
48      }
49  
50      public String getHost() {
51          return host;
52      }
53  
54      public String getEmailHost() {
55          return emailHost;
56      }
57  
58      public String getEmailSender() {
59          return emailSender;
60      }
61  
62      public String getEncKey() {
63          return encKey;
64      }

66      @Timed @ExceptionMetered
67      public boolean sendEmail(String fromUser, String replyTo, String to, String subject, String text) throws IOException {
68          Map<String, String> params = new HashMap<String, String>();
69          Enc enc = new Enc(encKey, 128);
70          params.put("from", enc.encrypt(fromUser));
71          params.put("replyTo", enc.encrypt(replyTo));
72          params.put("to", enc.encrypt(to));
73          // params.put("bcc", bcc);
74          params.put("subject", enc.encrypt(subject));
75          params.put("text", enc.encrypt(text));
76          // params.put("html", html);
77          String res = NetUtil.doPost(host + "/mailOut", params);
78          return res.startsWith("ok");
79      }
80  
81      @Timed @ExceptionMetered
82      public boolean sendXmppMessage(String from, String to, String message) throws IOException {
83          Map<String, String> params = new HashMap<String, String>();
84          Enc enc = new Enc(encKey, 128);
85          params.put("from", enc.encrypt(from));
86          params.put("to", enc.encrypt(to));
87          params.put("message", enc.encrypt(message));
88          String res = NetUtil.doPost(host + "/xmppOut", params);
89          if (!res.startsWith("ok")) {
90              logger.warn("Unable to send xmpp message, result = " + res);
91          }
92          return res.startsWith("ok");
93      }
