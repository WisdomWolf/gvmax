51      @Override
52      public void process(XMPPAction action) {
53          MetricsUtil.getCounter(XMPPReceiver.class, "received").inc();
54          Context timer = MetricsUtil.getTimer(XMPPReceiver.class, "processTime").time();
55          TimeTrack tt = new TimeTrack();
56          try {
57              tt.mark("received.time", action.getTimestamp());
58              tt.mark("process.time");
59              User user = extractInfo(action);
60              XMPPProcessor processor = new XMPPProcessor(user, action);
61              if (user == null) {
62                  MetricsUtil.getCounter(XMPPReceiver.class, "unregistered").inc();
63                  processor.sendMessage("You are not a registered gvmax user. Action cancelled");
64                  return;
65              }
66              userDAO.incrementGTalkCount(user.getEmail());
67              if (!processor.processCommands()) {
68                  if (!user.isGvPassword()) {
69                      processor.sendMessage("google voice password required to send SMS replies");
70                      return;
71                  }
72                  if ("unknown".equals(action.getBotNumber())) {
73                      processor.sendMessage("cannot send sms to unknown number");
74                      return;
75                  }
76                  try {
77                      Notifier notifier = new Notifier(user, userDAO, null);
78                      int num = notifier.sendSMS(action.getBotNumber(), action.getMessage());
79                      if (num > 1) {
80                          processor.sendMessage("Sending SMS to " + num + " numbers");
81                      }
82                  } catch (Exception e) {
83                      processor.sendMessage("Failed to send SMS error = " + e.getMessage());
84                  }
85              }
86          } finally {
87              timer.stop();
88              tt.mark("end.time");
89              logger.info("XMPP_IN ProcessTime: " + tt.deltaInSecs() + ", total: " + tt.elapsedInSecs());
90          }
91      }

93      private User extractInfo(XMPPAction action) {
94          String[] to = action.getTo();
95          String botJId = to[0].substring(6, to[0].indexOf('>'));
96          String botNumber = botJId.substring(0, botJId.indexOf('@'));
97          String senderJId = action.getFrom().substring(6, action.getFrom().indexOf('>'));
98          String gtalkEmail = EmailUtils.formatEmail(action.getFrom().substring(6, action.getFrom().indexOf('/')));
99          action.setBotJId(botJId);
100         action.setBotNumber(botNumber);
101         action.setSenderJId(senderJId);
102         return userDAO.retrieveByGTalk(gtalkEmail);
103     }

114         private boolean processCommands() {
115             String cmd = action.getMessage().trim().toLowerCase();
116             if (COMMAND_HELP.equals(cmd)) {
117                 help();
118                 return true;
119             }
120             if (COMMAND_INFO.equals(cmd)) {
121                 info();
122                 return true;
123             }
124             if (cmd.startsWith(COMMAND_ADDNUMBER)) {
125                 return addNumber();
126             }
127             if (COMMAND_PHONES.equals(cmd)) {
128                 displayPhones();
129                 return true;
130             }
131             if (cmd.startsWith(COMMAND_PHONES)) {
132                 return setFwdPhone();
133             }
134             if (COMMAND_CALL.equals(cmd)) {
135                 callNumber(action.getBotNumber());
136                 return true;
137             }
138             if (cmd.startsWith(COMMAND_CALL)) {
139                 String[] args = cmd.split(" ");
140                 if (args.length > 1) {
141                     callNumber(restOf(args));
142                     return true;
143                 }
144             }
145             if (COMMAND_GROUPS.equals(cmd)) {
146                 showGroups();
147                 return true;
148             }
149             if (cmd.startsWith(COMMAND_GROUPLIST)) {
150                 String[] args = cmd.split(" ");
151                 if (args.length == 2) {
152                     try {
153                         int index = Integer.parseInt(args[1]);
154                         listGroup(index);
155                         return true;
156                     } catch (Exception e) {
157                         MiscUtils.emptyBlock();
158                     }
159                 }
160             }
161             if (cmd.startsWith(COMMAND_GROUPS)) {
162                 String[] args = cmd.split(" ");
163                 if (args.length == 2) {
164                     try {
165                         int index = Integer.parseInt(args[1]);
166                         addGroup(index);
167                         return true;
168                     } catch (Exception e) {
169                         MiscUtils.emptyBlock();
170                     }
171                 }
172             }
173             return false;
174         }

195         private void info() {
196             sendMessage("retrieving info for " + action.getBotNumber() + "...");
197             ContactEntry entry = ContactUtil.getContactEntry(user, action.getBotNumber());
198             StringBuffer msg = new StringBuffer();
199             if (entry != null) {
200                 if (entry.hasName()) {
201                     Name name = entry.getName();
202                     if (name.hasFullName()) {
203                         String fullNameToDisplay = name.getFullName().getValue();
204                         if (name.getFullName().hasYomi()) {
205                             fullNameToDisplay += " (" + name.getFullName().getYomi() + ")";
206                         }
207                         msg.append("Name: " + fullNameToDisplay + "\n");
208                     }
209                 }
210                 if (entry.getEmailAddresses().size() > 0) {
211                     msg.append("Email Addresses:\n");
212                     for (Email email : entry.getEmailAddresses()) {
213                         msg.append("    " + email.getAddress() + "\n");
214                     }
215                 }
216                 if (entry.getPhoneNumbers().size() > 0) {
217                     msg.append("Phone Numbers:\n");
218                     for (PhoneNumber number : entry.getPhoneNumbers()) {
219                         msg.append("    ");
220                         msg.append(PhoneUtil.normalizeNumber(number.getPhoneNumber()));
221                         if (number.getLabel() != null) {
222                             msg.append(" [" + number.getLabel().substring(number.getLabel().lastIndexOf('#') + 1) + "]");
223                         }
224                         if (number.getRel() != null) {
225                             msg.append(" [" + number.getRel().substring(number.getRel().lastIndexOf('#') + 1) + "]");
226                         }
227                         msg.append("\n");
228                     }
229                 }
230             } else {
231                 msg.append("Contact information for " + action.getBotNumber() + " could not retrieved.");
232             }
233             sendMessage(msg.toString());
234         }

236         private boolean addNumber() {
237             String[] info = action.getMessage().split(" ");
238             if (info.length <= 1) {
239                 return false;
240             }
241             String number = PhoneUtil.normalizeNumber(restOf(info));
242             if (!PhoneUtil.isNumber(number)) {
243                 return false;
244             }
245             sendMessage("Adding buddy " + number);
246             String nickname = ContactUtil.getContact(user, number);
247             if (nickname == null) {
248                 nickname = number;
249             }
250             try {
251                 GTalk.addBuddy(user.getgTalkEmail(), user.getgTalkPassword(), number + "@" + relay.getAppName() + ".appspotchat.com", nickname, user.getgTalkGroup());
252             } catch (IOException e) {
253                 sendMessage("Unable to add number " + number + " : " + e.getMessage());
254             }
255             sendMessage("Added " + number + " as " + nickname);
256             return true;
257         }

259         private void displayPhones() {
260             if (!user.isGvPassword()) {
261                 sendMessage("google voice password required for this functionality");
262                 return;
263             }
264             sendMessage("retrieving available forward phones...");
265             StringBuffer msg = new StringBuffer();
266             try {
267                 GoogleVoice gv = new GoogleVoice(user.getEmail(), user.getPassword());
268                 List<GoogleVoicePhone> phones = gv.getPhones();
269                 if (phones.size() == 0) {
270                     msg.append("No phones have been registered with your GoogleVoice account");
271                 } else {
272                     int pn = 1;
273                     for (GoogleVoicePhone phone : phones) {
274                         if (PhoneUtil.numbersMatch(phone.getNumber(), user.getGvFwdPhone())) {
275                             msg.append(pn + ". ** " + phone.getName() + "  " + PhoneUtil.normalizeNumber(phone.getNumber()) + " **\n");
276                         } else {
277                             msg.append(pn + ". " + phone.getName() + "  " + PhoneUtil.normalizeNumber(phone.getNumber()) + "\n");
278                         }
279                         pn += 1;
280                     }
281                 }
282                 sendMessage(msg.toString());
283             } catch (Exception e) {
284                 sendMessage("Unable to retrieve phones : " + e.getMessage());
285             }
286         }

288         private boolean setFwdPhone() {
289             if (!user.isGvPassword()) {
290                 sendMessage("google voice password required for this functionality");
291                 return true;
292             }
293 
294             String[] info = action.getMessage().split(" ");
295             if (info.length != 2) {
296                 return false;
297             }
298             int index = -1;
299             try {
300                 index = Integer.parseInt(info[1]);
301             } catch (Exception e) {
302                 return false;
303             }
304 
305             try {
306                 GoogleVoice gv = new GoogleVoice(user.getEmail(), user.getPassword());
307                 List<GoogleVoicePhone> phones = gv.getPhones();
308                 if (phones.size() == 0) {
309                     sendMessage("No phones have been registered with your GoogleVoice account");
310                 } else {
311                     if (index <= 0) {
312                         sendMessage("Invalid phone index '" + index + "' please select an index from 1 to " + phones.size());
313                         return true;
314                     }
315                     if (phones.size() < index) {
316                         sendMessage("invalid phone index '" + index + "' please select an index from 1 to " + phones.size());
317                         displayPhones();
318                         return true;
319                     }
320                     GoogleVoicePhone phone = phones.get(index - 1);
321                     // HANI new API
322                     userDAO.setGVFwdPhone(user.getEmail(), phone.getNumber(), phone.getType());
323                     sendMessage("Foward phone set to : " + phone.getName());
324                 }
325             } catch (Exception e) {
326                 sendMessage("Unable to select phone: " + e.getMessage());
327             }
328             return true;
329         }

331         private void callNumber(String number) {
332             if (!user.isGvPassword()) {
333                 sendMessage("google voice password required for this functionality");
334                 return;
335             }
336             if (user.getGvFwdPhone() == null) {
337                 sendMessage("forwarding phone not selected, please select phone via ggp command");
338                 return;
339             }
340             number = PhoneUtil.normalizeNumber(number);
341             if (!PhoneUtil.isNumber(number)) {
342                 sendMessage(number + " is not valid");
343                 return;
344             }
345 
346             GoogleVoice gv = new GoogleVoice(user.getEmail(), user.getPassword());
347             sendMessage("Placing call to " + number + ", " + user.getGvFwdPhone() + " set as fowarding phone");
348             try {
349                 gv.call(number, user.getGvFwdPhone());
350             } catch (IOException e) {
351                 sendMessage("Call failed : " + e.getMessage());
352             }

377         private void addGroup(int index) {
378             try {
379                 sendMessage("adding group...");
380                 List<String> groups = ContactUtil.getGroups(user);
381                 if (index <= 0) {
382                     sendMessage("Invalid group index '" + index + "' please select an index from 1 to " + groups.size());
383                     showGroups();
384                     return;
385                 }
386                 if (groups.size() < index) {
387                     sendMessage("invalid group index '" + index + "' please select an index from 1 to " + groups.size());
388                     showGroups();
389                     return;
390                 }
391                 String group = groups.get(index - 1);
392                 String g = group;
393                 g = g.replaceAll(" ", "_");
394                 g = g.replaceAll(":", ".");
395                 GTalk.addBuddy(user.getgTalkEmail(), user.getgTalkPassword(), g + "@" + relay.getAppName() + ".appspotchat.com", group, user.getgTalkGroup());
396                 sendMessage("group " + group + " added");
397             } catch (IOException e) {
398                 sendMessage("unable to add group : " + e.getMessage());
399             }
400         }

402         private void listGroup(int index) {
403             sendMessage("retrieving group members");
404             try {
405                 List<String> groups = ContactUtil.getGroups(user);
406                 if (index <= 0) {
407                     sendMessage("Invalid group index '" + index + "' please select an index from 1 to " + groups.size());
408                     showGroups();
409                     return;
410                 }
411                 if (groups.size() < index) {
412                     sendMessage("invalid group index '" + index + "' please select an index from 1 to " + groups.size());
413                     showGroups();
414                     return;
415                 }
416                 String group = groups.get(index - 1);
417                 Map<String, String> contacts = ContactUtil.getNumbersInGroup(user, group);
418                 if (contacts.size() == 0) {
419                     sendMessage("group " + group + " has no members");
420                     return;
421                 }
422                 StringBuffer msg = new StringBuffer("Contacts for " + group + "\n");
423                 for (Map.Entry<String, String> contact : contacts.entrySet()) {
424                     msg.append(contact.getKey() + " : " + contact.getValue() + "\n");
425                 }
426                 msg.append("Total : ");
427                 msg.append(contacts.size());
428                 sendMessage(msg.toString());
429 
430             } catch (IOException e) {
431                 sendMessage("Unable to get group contacts: " + e.getMessage());
432             }
