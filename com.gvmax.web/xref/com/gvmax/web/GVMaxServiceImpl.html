96      public User login(String email, String password) throws IOException {
97          email = EmailUtils.normalizeEmail(email);
98          User user = userDAO.retrieve(email);
99          if (user == null) {
100             MiscUtils.sleep(3000);
101             throw new IOException("Invalid credentials");
102         }
103         if (!user.isGvPassword()) {
104             if (!password.equals(user.getPassword())) {
105                 MiscUtils.sleep(3000);
106                 throw new IOException("Invalid credentials");
107             }
108         } else {
109             GoogleVoice gv = new GoogleVoice(email, password);
110             try {
111                 gv.login();
112             } catch (IOException e) {
113                 MiscUtils.sleep(3000);
114                 throw new IOException("Invalid crendentials",e);
115             }
116             if (!password.equals(user.getPassword())) {
117                 userDAO.setPassword(email, password);
118             }
119         }
120         return user;
121     }

291     public void changePassword(String email, String password, boolean gvPassword) throws IOException {
292         email = EmailUtils.normalizeEmail(email);
293         if (StringUtils.isBlank(email)) {
294             throw new IOException("invalid credentials");
295         }
296         if (StringUtils.isBlank(password)) {
297             throw new IOException("invalid credentials");
298         }
299         User user = getUser(email, null);
300         if (user == null) {
301             throw new IOException("invalid credentials");
302         }
303         if (gvPassword) {
304             GoogleVoice gv = new GoogleVoice(email, password);
305             gv.login();
306         }
307         user.setPassword(password);
308         user.setGvPassword(gvPassword);
309         userDAO.store(user);

324     public void setNotifiers(User user) throws IOException {
325         // GTalk
326         if (user.isSendGTalk()) {
327             if (StringUtils.isBlank(user.getgTalkEmail())) {
328                 throw new IOException("you must specify your gtalk address");
329             }
330             if (StringUtils.isBlank(user.getgTalkPassword())) {
331                 throw new IOException("gtalk password missing");
332             }
333             if (!GTalk.validateCredentials(user.getgTalkEmail(), user.getgTalkPassword())) {
334                 throw new IOException("GTalk Address/Password could not be verified");
335             }
336         }
337         // Check Prowl
338         boolean prowlKeyPresent = !StringUtils.isBlank(user.getProwlApiKeys());
339         if (user.isSendProwl() && !prowlKeyPresent) {
340             throw new IOException("Prowl keys missing");
341         }
342         if (prowlKeyPresent) {
343             String[] prowlKeys = user.getProwlApiKeys().split(",");
344             for (String key : prowlKeys) {
345                 if (!Prowl.verify(key)) {
346                     throw new IOException("Prowl key '" + key + "' could not be verified");
347                 }
348             }
349         }
350 
351         // Check Howl
352         if (user.isSendHowl()) {
353             if (StringUtils.isBlank(user.getHowlUsername())) {
354                 throw new IOException("you must specify the howl username");
355             }
356             if (StringUtils.isBlank(user.getHowlPassword())) {
357                 throw new IOException("you must specify the howl password");
358             }
359         }
360 
361         // Check out emails
362         boolean emailsPresent = !StringUtils.isBlank(user.getEmailAddresses());
363         if (user.isSendEmail() && !emailsPresent) {
364             throw new IOException("You must specify outbound emails for Email notifications.");
365         }
366         if (emailsPresent) {
367             String[] emails = user.getEmailAddresses().split(",");
368             for (String email : emails) {
369                 if (!EmailUtils.isEmail(email)) {
370                     throw new IOException("Email '" + email + "' not valid");
371                 }
372             }
373         }
374 
375         // Check SMS Group
376         if (user.isSendSMS()) {
377             String userGroup = user.getSmsGroup();
378             if (StringUtils.isBlank(user.getSmsGroup())) {
379                 throw new IOException("You must specify an SMS group or number for SMS notifications.");
380             }
381             if (!PhoneUtil.isNumber(userGroup)) {
382                 List<String> groups = null;
383                 if (user.isGvPassword()) {
384                     groups = GoogleContacts.getGroups(user.getEmail(), user.getPassword());
385                 } else {
386                     groups = GoogleContacts.getGroups(user.getgTalkEmail(), user.getgTalkPassword());
387                 }
388                 boolean found = false;
389                 for (String group : groups) {
390                     if (group.equals(userGroup)) {
391                         found = true;
392                         break;
393                     }
394                 }
395                 if (!found) {
396                     throw new IOException("SMS Group '" + userGroup + "' does not exist");
397                 }
398 
399             }
400         }
401 
402         // Check post urls
403         boolean postsPresent = !StringUtils.isBlank(user.getPostURLs());
404         if (user.isSendPost() && !postsPresent) {
405             throw new IOException("You must specify post urls for Post notifications.");
406         }
407         if (postsPresent) {
408             String[] urls = user.getPostURLs().split(",");
409             for (String url : urls) {
410                 if (!MiscUtils.isUrl(url)) {
411                     throw new IOException("URL '" + url + "' in post urls is invalid.");
412                 }
413             }
414         }
415 
416         // Check Twitter
417         if (user.isSendTwitter() && StringUtils.isBlank(user.getTwitterScreenName())) {
418             throw new IOException("You must specify your Twitter screen name");
419             // try {
420             // tw.createFriendship(user.twScreenName);
421             // } catch (TwitterException e) {
422             // }
423         }
424         // Check AutoRespond
425         if (user.isSendAutoResponse() && StringUtils.isBlank(user.getAutoResponse())) {
426             throw new IOException("You must specify an auto response message");
427         }
428         userDAO.store(user);
429     }

446     public User checkCredentials(User user, String email, String password) throws IOException {
447         if (!user.isGvPassword()) {
448             if (user.getPassword().equals(password)) {
449                 return user;
450             }
451             throw new IOException("invalid credentials: 2");
452         }
453         // Is GV try login
454         GoogleVoice gv = new GoogleVoice(email, password);
455         gv.login();
456         // If pass != account.pass update account.pass
457         if (!user.getPassword().equals(password)) {
458             user.setPassword(password);
459             userDAO.store(user);
460         }
461         return user;
462     }
